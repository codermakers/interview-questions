#### 4、手动实现 Promise.all/Promise.race/Promise.any/Promise.myAllSettled

```js
// Promise.all
function myPromiseAll(promises) {
  return new Promise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      reject(new TypeError('Argument must be an array'));
      return;
    }
    if (promises.length === 0) {
      resolve([]);
      return;
    }
    const results = new Array(promises.length);
    let completed = 0;
    promises.forEach((promise, index) => {
      Promise.resolve(promise)
        .then((value) => {
          results[index] = value;
          completed++;
          if (completed === promises.length) resolve(results);
        })
        .catch((err) => reject(err));
    });
  });
}

// Promise.race
function myPromiseRace(promises) {
  return new Promise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      reject(new TypeError('Argument must be an array'));
      return;
    }
    promises.forEach((promise) => {
      Promise.resolve(promise).then(resolve).catch(reject);
    });
  });
}
```

#### 7、实现一个发布订阅模式：

```js
/**
 * 发布订阅模式（Pub/Sub）核心类
 * 核心功能：事件订阅、事件发布、取消订阅、清空所有订阅
 * 边界处理：重复订阅、取消不存在的订阅、发布时无订阅者、参数透传
 */
class EventEmitter {
  constructor() {
    // 存储事件与订阅者的映射关系：{ eventName: [callback1, callback2, ...] }
    this.events = Object.create(null); // 用Object.create(null)避免原型链污染
  }

  on(eventName, callback, once = false) {
    // 边界1：校验参数合法性
    if (typeof eventName !== 'string' || typeof callback !== 'function') {
      throw new Error('事件名称必须为字符串，回调必须为函数');
    }

    // 初始化事件队列：不存在则创建空数组
    if (!this.events[eventName]) {
      this.events[eventName] = [];
    }

    // 边界2：避免重复订阅同一个回调（可选，根据业务需求决定是否开启）
    const isDuplicate = this.events[eventName].some(
      (item) => item.callback === callback,
    );
    if (isDuplicate) {
      console.warn(`事件${eventName}已订阅过该回调，无需重复订阅`);
      return this;
    }

    // 存储回调+是否只执行一次的标记
    this.events[eventName].push({ callback, once });
    return this; // 链式调用
  }

  once(eventName, callback) {
    return this.on(eventName, callback, true);
  }

  emit(eventName, ...args) {
    // 边界3：事件无订阅者时直接返回
    if (!this.events[eventName] || this.events[eventName].length === 0) {
      console.warn(`事件${eventName}暂无订阅者`);
      return this;
    }

    // 遍历执行所有订阅回调（拷贝数组避免执行过程中数组长度变化）
    const callbacks = [...this.events[eventName]];
    callbacks.forEach((item, index) => {
      try {
        // 透传参数给回调函数
        item.callback.apply(this, args);
      } catch (error) {
        // 边界4：捕获回调执行错误，避免影响其他回调
        console.error(`事件${eventName}的回调执行失败：`, error);
      }

      // 一次性事件：执行后移除
      if (item.once) {
        this.events[eventName].splice(index, 1);
      }
    });

    return this;
  }

  off(eventName, callback) {
    // 边界5：事件无订阅者时直接返回
    if (!this.events[eventName]) {
      console.warn(`事件${eventName}暂无订阅者，无需取消`);
      return this;
    }

    // 过滤掉要取消的回调
    this.events[eventName] = this.events[eventName].filter(
      (item) => item.callback !== callback,
    );

    // 事件无订阅者时清空该事件（优化内存）
    if (this.events[eventName].length === 0) {
      delete this.events[eventName];
    }

    return this;
  }

  clear(eventName) {
    if (eventName) {
      // 清空指定事件
      delete this.events[eventName];
    } else {
      // 清空所有事件
      this.events = Object.create(null);
    }
    return this;
  }

  getListenerCount(eventName) {
    return this.events[eventName] ? this.events[eventName].length : 0;
  }
}
```
